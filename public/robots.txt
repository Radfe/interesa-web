<?php
declare(strict_types=1);

/**
 * Univerzálna šablóna článku.
 * Očakáva, že volajúca stránka nastaví $ARTICLE_SLUG (napr. v /clanky/xyz.php).
 * Načíta meta z inc/articles.php (+ inc/articles_ext.php, ak existuje)
 * a HTML obsah z /content/articles/<slug>.html
 */

$ROOT = __DIR__;

// Základné include-y (hlavička/layout)
require_once $ROOT . '/inc/head.php';

// --- Pomocné funkcie (lokálne, aby si nemusel meniť inc/functions.php) ---
if (!function_exists('article_get')) {
  /**
   * @return array{0:string,1:string,2:string}|null [title, perex, category] alebo null
   */
  function article_get(string $slug): ?array {
    // $ART sa doplní nižšie načítaním inc/articles(.php|_ext.php)
    return isset($GLOBALS['ART'][$slug]) ? $GLOBALS['ART'][$slug] : null;
  }
}
if (!function_exists('article_content_path')) {
  function article_content_path(string $slug): string {
    return __DIR__ . '/content/articles/' . $slug . '.html';
  }
}

// --- Načítaj register článkov ($ART) ---
if (!isset($ART) || !is_array($ART)) {
  $ART = [];
}
$articles_php = $ROOT . '/inc/articles.php';
if (is_file($articles_php)) {
  /** @noinspection PhpIncludeInspection */
  require $articles_php; // očakáva sa, že nastaví (alebo rozšíri) $ART
}
$articles_ext = $ROOT . '/inc/articles_ext.php';
if (is_file($articles_ext)) {
  /** @noinspection PhpIncludeInspection */
  require $articles_ext; // bezpečné doplnenie $ART bez zásahu do core
}

// --- Urči slug článku ---
$ARTICLE_SLUG = isset($ARTICLE_SLUG) ? (string)$ARTICLE_SLUG : '';
if ($ARTICLE_SLUG === '' && isset($_GET['slug'])) {
  $ARTICLE_SLUG = basename((string)$_GET['slug']);
}
if ($ARTICLE_SLUG === '') {
  http_response_code(404);
  echo '<section class="container"><h1>Článok sa nenašiel</h1><p>Slug chýba.</p></section>';
  require $ROOT . '/inc/footer.php';
  exit;
}

// --- Meta a obsah ---
$meta = article_get($ARTICLE_SLUG);
// [title, perex, category]
$title    = $meta[0] ?? ucwords(str_replace('-', ' ', $ARTICLE_SLUG));
$perex    = $meta[1] ?? '';
$category = $meta[2] ?? '';

$content_path = article_content_path($ARTICLE_SLUG);
if (!is_file($content_path)) {
  http_response_code(404);
  echo '<section class="container"><h1>Článok sa nenašiel</h1><p>Chýba obsahový súbor: <code>/content/articles/' . htmlspecialchars($ARTICLE_SLUG, ENT_QUOTES) . '.html</code>.</p></section>';
  require $ROOT . '/inc/footer.php';
  exit;
}

// SEO/canonical (ak tvoja head.php používa tieto premené, super; ak nie, nič sa nedeje)
$PAGE_TITLE       = $title . ' | Interesa.sk';
$PAGE_DESCRIPTION = $perex ?: 'Nezávislý článok Interesa.sk';
$PAGE_CANONICAL   = '/clanky/' . $ARTICLE_SLUG . '.php';

// --- Render ---
?>
<section class="content-main">
  <div class="container">
    <nav class="breadcrumbs" aria-label="Drobená navigácia" style="margin: 8px 0 16px;">
      <a href="/">Domov</a> › <a href="/clanky/">Články</a> › <span><?= htmlspecialchars($title, ENT_QUOTES) ?></span>
    </nav>

    <header class="article-header" style="margin-bottom: 16px;">
      <h1 style="margin: 0 0 8px;"><?= htmlspecialchars($title, ENT_QUOTES) ?></h1>
      <?php if ($perex !== ''): ?>
        <p style="color:#6b7280;max-width:65ch;margin:0;"><?= htmlspecialchars($perex, ENT_QUOTES) ?></p>
      <?php endif; ?>
      <?php if ($category !== ''): ?>
        <div style="margin-top:8px;"><span style="background:#eef6ff;color:#1163c2;padding:4px 10px;border-radius:999px;font-size:.9rem;">Kategória: <?= htmlspecialchars($category, ENT_QUOTES) ?></span></div>
      <?php endif; ?>
    </header>

    <article class="article-body" style="max-width: 900px;">
      <?php
        // Vypíš HTML obsah článku
        // (obsah je bezpečne pripravený v repozitári; ak by si sem chcel HTML sanitizáciu, doplníme)
        readfile($content_path);
      ?>
    </article>
  </div>
</section>
<?php
require $ROOT . '/inc/footer.php';
